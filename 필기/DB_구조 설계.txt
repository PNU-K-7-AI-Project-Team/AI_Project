Users 테이블

CREATE TABLE IF NOT EXISTS Users (
    UserCode VARCHAR(50) PRIMARY KEY,       -- 사용자의 고유 코드
    UserName VARCHAR(100) NOT NULL,         -- 사용자 이름
    Position VARCHAR(100),                  -- 사용자 직위
    Contact VARCHAR(20),                    -- 연락처
    Department VARCHAR(100),                -- 소속 부서
    Region VARCHAR(100),                    -- 지역
    Status ENUM('Active', 'Inactive', 'On Leave', 'Overworked', 'Stressed', 'Fatigued', 'At Risk', 'Emergency', 'Recovering')
    DateOfBirth DATE,                       -- 생년월일
    Gender ENUM('M', 'F'),                  -- 성별 (M: 남성, F: 여성)
    CreatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP,  -- 생성일자 (레코드 생성 시 자동 설정)
    UpdatedAt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP  -- 업데이트일자 (레코드 업데이트 시 자동 설정)
);

각 상태의 사용 예:
Active: 작업자가 정상적으로 작업 중일 때, 위험 요소가 없는 상태입니다.
Inactive: 작업자가 쉬고 있거나 현재 작업 중이지 않은 상태를 나타냅니다.
On Leave: 휴가 중인 작업자, 작업 현장에 부재한 상태입니다.
Overworked: 데이터 분석을 통해 작업자가 과도한 작업으로 인해 위험 상태에 있는 것으로 감지된 경우입니다.
Stressed: 심박수 및 기타 생체 데이터가 스트레스 상태를 나타내는 경우 사용됩니다.
Fatigued: 작업자가 피로 상태에 빠진 것으로 분석된 경우 사용됩니다.
At Risk: 비정상적인 데이터가 감지되어 작업자가 위험에 처한 것으로 간주되는 상태입니다.
Emergency: 즉각적인 조치가 필요한 긴급 상황을 나타냅니다.
Recovering: 작업자가 이전에 위험 상태였지만, 현재는 회복 중인 상태입니다.



생체 데이터 테이블 (BiometricData)

CREATE TABLE IF NOT EXISTS BiometricData (
    id INT AUTO_INCREMENT PRIMARY KEY,
    WorkDate DATE NOT NULL,                -- 작업 날짜
    UserCode INT NOT NULL,                 -- 사용자 코드 (INT로 처리)
    Heartbeat INT,                         -- 심박수 (BPM)
    Temperature FLOAT,                     -- 체온 (도)
    Spo2 INT,                              -- 산소포화도 (%)
    Latitude DECIMAL(10, 7),               -- 위도 (정밀한 좌표)
    Longitude DECIMAL(10, 7),              -- 경도 (정밀한 좌표)
    VitalDate TIME                         -- 생체 데이터 수집 시간
);


자이로 데이터 테이블 (GyroData)

CREATE TABLE IF NOT EXISTS GyroData (
    id INT AUTO_INCREMENT PRIMARY KEY,
    WorkDate DATE NOT NULL,                -- 작업 날짜
    UserCode INT NOT NULL,                 -- 사용자 코드 (INT로 처리)
    X FLOAT,                               -- X축 데이터
    Y FLOAT,                               -- Y축 데이터
    Z FLOAT,                               -- Z축 데이터
    RegisterDate DATETIME                  -- 자이로 데이터 기록 시간 (날짜 및 시간 포함)
);
